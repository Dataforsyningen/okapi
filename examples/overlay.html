        overlays: [
          {
            name: 'dagi',
            type: 'WMS',
            title: 'Kommune grænser',
            visible: true,
            service: 'dagi',
            layer: 'kommune',
            style: 'red',
            format: 'image/png',
            source: 'kf'
          }
        ]
      });

      const checkbox = document.getElementById('checkbox')
      checkbox.addEventListener('change', function (e) {
        maps[0].toggleLayer('dagi', this.checked)
      })
