        <div>
          <input type="checkbox" id="checkbox" checked>
          <label for="checkbox">
            Stednavne
          </label>
        </div>

        <script>
          var maps = new okapi.Initialize({
            overlays: [
              {
                name: 'dhm',
                type: 'WMS',
                title: 'Stednavne',
                visible: true,
                service: 'forvaltning2',
                layer: 'Stednavne_basiskort',
                format: 'image/png',
                source: 'kf'
              }
            ]
          });

          const checkbox = document.getElementById('checkbox');
          checkbox.addEventListener('change', function (e) {
            maps[0].toggleLayer('dhm', this.checked)
          });
        </script>
